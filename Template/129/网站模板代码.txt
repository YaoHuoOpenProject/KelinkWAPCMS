			$(hour_elem).text(hour<10?"0"+hour:hour);//计算小时
			$(minute_elem).text(minute<10?"0"+minute:minute);//计算分

		} else { 
		  nMsDjs=0;

			clearInterval(timer);
		}
	}, 1000);
